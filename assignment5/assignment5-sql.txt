SELECT o.order_id, 
       SUM(li.quantity * p.price) AS total_price 
FROM orders o 
       JOIN line_items li ON o.order_id = li.order_id 
       JOIN products p ON li.product_id = p.product_id 
GROUP BY o.order_id 
ORDER BY o.order_id ASC 
LIMIT 5;

SELECT c.customer_name, AVG(t.total_price) AS average_order_price FROM customers c JOIN ( SELECT o.customer_id, SUM(li.quantity*p.price) as total_price  FROM orders o JOIN line_items li ON o.order_id = li.order_id JOIN products p ON li.product_id = p.product_id GROUP BY o.order_id ) AS t ON c.customer_id = t.customer_id GROUP BY c.customer_id, c.customer_name ORDER BY c.customer_name;

SELECT e.first_name, 
       e.last_name, 
       COUNT(o.order_id) AS order_count 
FROM employees e 
       LEFT JOIN orders o ON e.employee_id = o.employee_id 
GROUP BY e.employee_id, e.first_name, e.last_name 
HAVING COUNT(o.order_id) > 5 
ORDER BY e.last_name;
